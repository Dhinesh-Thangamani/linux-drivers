#!/bin/bash
set -e

# Create directory
sudo mkdir -p /opt/reboot-cycle

# Copy reboot script
sudo cp reboot_cycle.sh /opt/reboot-cycle/

# Make the script executable
sudo chmod +x /opt/reboot-cycle/reboot_cycle.sh

# Copy systemd service file
sudo cp reboot-cycle.service /etc/systemd/system/

# Reload systemd
sudo systemctl daemon-reload

# Enable the service
sudo systemctl enable reboot-cycle.service

# Start the reboot cycle service
sudo systemctl start reboot-cycle.service

# Optional checks
echo "Reboot cycle setup complete."
echo "Count (if exists):"
sudo cat /opt/reboot-cycle/count 2>/dev/null || echo "Count file not found."

echo "Log (if exists):"
sudo cat /opt/reboot-cycle/reboot.log 2>/dev/null || echo "Log file not found."
